00169     def load_fuse_kafka_event(self, string):
00170         """ Decodes a json logstash/fuse_kafka event string, i.e.:
00171                 - does a json decoding
00172                 - decodes @message and command fields 
00173                 - adds message_size-added field 
00174         
00175         string - the JSON input logstash/fuse_kafka event
00176 
00177         Example
00178             
00179             build.FuseKafkaLog().load_fuse_kafka_event(
00180               '{"command": "bXkgY29tbWFuZGU=", "@message": "bXkgbWVzc2FnZQ=="}')
00181 
00182             => {'message_size-added': 10,
00183                 u'command': 'my commande',
00184                 u'@message': 'my message'}
00185 
00186         Returns the decoded json object
00187         """
00188         event = json.loads(string)
00189         for item in ["@message", "command"]:
00190             event[item] += "=" * ((4 - len(event[item]) % 4) % 4)
00191             event[item] = base64.b64decode(event[item] + "==")
00192         event["message_size-added"] = len(event["@message"])
00193         return event
